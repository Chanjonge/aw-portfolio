【2단계】 브라우저에서 Seed API 실행
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

https://aw-portfolio-chi.vercel.app 페이지에서
F12 (개발자 도구) → Console 탭에 아래 코드 붙여넣기:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
fetch('https://aw-portfolio-chi.vercel.app/api/seed', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ 
    secretKey: 'default-seed-key-2025' 
  })
})
.then(r => r.json())
.then(data => {
  console.log('✅ Seed 완료!', data);
  alert('데이터 생성 완료! 이제 로그인하세요.');
})
.catch(err => {
  console.error('❌ 오류:', err);
  alert('오류 발생: ' + err.message);
})
